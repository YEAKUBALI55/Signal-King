// EMA Calculation Logic function calculateEMA(data, period) { let k = 2 / (period + 1); let ema = data[0]; for (let i = 1; i < data.length; i++) { ema = data[i] * k + ema * (1 - k); } return ema; } // ATR Calculation Logic function calculateATR(data, period = 14) { let trs = []; for (let i = 1; i < data.length; i++) { let high = parseFloat(data[i][2]); let low = parseFloat(data[i][3]); let prevClose = parseFloat(data[i - 1][4]); trs.push(Math.max(high - low, Math.abs(high - prevClose), Math.abs(low - prevClose))); } return trs.slice(-period).reduce((a, b) => a + b) / period; } // Main Analysis Function async function analyzeConditions(coin) { const klines = await fetchKlines(coin, '1d', 200); const closePrices = klines.map(k => parseFloat(k[4])); // ইন্ডিকেটর মান const ema20 = calculateEMA(closePrices, 20); const ema50 = calculateEMA(closePrices, 50); const ema200 = calculateEMA(closePrices, 200); const atr = calculateATR(klines, 14); const last = klines[klines.length - 1]; const prev = klines[klines.length - 2]; const price = parseFloat(last[4]); const high = parseFloat(last[2]); const low = parseFloat(last[3]); // ১. Trend Filter const isTrendBullish = (ema20 > ema50 && ema50 > ema200 && price > ema20); // ২. RSI Filter (Simplified) const rsi = 60; // এখানে আপনার RSI ক্যালকুলেশন ফাংশন বসবে const isRsiValid = (rsi >= 55 && rsi <= 70); // ৩. Entry Logic (0.15% Breakout) const entryPrice = parseFloat(prev[2]) * 1.0015; const isBreakout = price >= entryPrice; // ৪. Risk Management (ATR Based) const stopLoss = parseFloat(prev[3]) - (atr * 1.5); // ATR buffer const tp3 = price + ((price - stopLoss) * 3); // 1:3 RR // ৫. Confidence Score let score = 0; if (parseFloat(prev[4]) > parseFloat(prev[1])) score += 20; // Y-Green if (price > parseFloat(last[1])) score += 20; // T-Green if (isTrendBullish) score += 30; if (isRsiValid) score += 20; if (parseFloat(last[5]) > 1000) score += 10; // Volume return { isBuy: (isTrendBullish && isRsiValid && isBreakout), score, entry: entryPrice, sl: stopLoss, tp3: tp3 }; }
