function runQuantumEngine(coin, price, percentChange) { const config = coinData[coin]; config.ema9 = (price * 0.2) + (config.ema9 * 0.8 || price); config.ema21 = (price * 0.095) + (config.ema21 * 0.905 || price); document.getElementById(`price-${coin}`).innerText = '$' + price.toFixed(config.decimals); // ১. যদি ট্রেড রানিং থাকে, তাহলে শুধু মনিটর করো (নতুন এন্ট্রি ক্যালকুলেট হবে না) if (config.currentTrackState === 'active_in_trade') { monitorTrade(coin, price); return; } // ২. যদি এন্ট্রি পেন্ডিং থাকে, তাহলে শুধু এন্ট্রি হিট হয়েছে কিনা চেক করো if (config.currentTrackState === 'monitoring_entry') { const isHit = config.direction === 'LONG' ? (price <= config.entryVal) : (price >= config.entryVal); if (isHit) { config.currentTrackState = 'active_in_trade'; document.getElementById(`card-${coin}`).className = "signal-card card-entry-triggered"; document.getElementById(`alert-badge-${coin}`).innerText = "ENTRY HIT"; } return; } // ৩. নতুন সিগন্যাল জেনারেটর (এই অংশটি শুধুমাত্র তখনই কাজ করবে যখন কোনো ট্রেড ওপেন নেই) const bTrendUp = config.ema9 > config.ema21; // শুধুমাত্র যদি currentTrackState 'standby' থাকে, তবেই নতুন সিগন্যাল জেনারেট করো if (config.currentTrackState === 'standby' && Math.abs(percentChange) > 0.1) { const signal = bTrendUp ? 'STRONG BUY' : 'STRONG SELL'; config.lastSignal = signal; config.direction = bTrendUp ? 'LONG' : 'SHORT'; config.currentTrackState = 'monitoring_entry'; // এন্ট্রি এবং টার্গেট লেভেল ফিক্সড করে দেওয়া হলো config.entryVal = bTrendUp ? price * 0.999 : price * 1.001; config.t1Val = bTrendUp ? price * 1.005 : price * 0.995; config.t2Val = bTrendUp ? price * 1.012 : price * 0.988; config.slVal = bTrendUp ? price * 0.993 : price * 1.007; // UI আপডেট const calcPct = (val) => (((val - config.entryVal) / config.entryVal) * 100).toFixed(2); document.getElementById(`entry-${coin}`).innerText = config.entryVal.toFixed(config.decimals); document.getElementById(`target1-${coin}`).innerText = config.t1Val.toFixed(config.decimals); document.getElementById(`p-t1-${coin}`).innerText = calcPct(config.t1Val) + '%'; document.getElementById(`target2-${coin}`).innerText = config.t2Val.toFixed(config.decimals); document.getElementById(`p-t2-${coin}`).innerText = calcPct(config.t2Val) + '%'; document.getElementById(`stop-${coin}`).innerText = config.slVal.toFixed(config.decimals); document.getElementById(`p-sl-${coin}`).innerText = calcPct(config.slVal) + '%'; document.getElementById(`signal-badge-${coin}`).innerText = signal; } }
