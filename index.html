function renderSignalCard(symbol, price, percentChange) { const data = coinData[symbol]; if (!data) return; const isLong = data.direction === 'LONG'; const container = document.getElementById('dynamic-cards-container'); if (!container) return; // Safety check to ensure container exists const card = document.createElement('div'); card.className = "signal-card"; card.id = `card-${symbol}`; card.innerHTML = `
${symbol}/USDT ${data.lastSignal}
AWAITING ENTRY
$${price.toFixed(data.decimals)}
Entry ${data.entryVal.toFixed(data.decimals)}
Target ${data.t1Val.toFixed(data.decimals)}
`; container.appendChild(card); }
